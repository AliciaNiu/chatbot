qdrant:
pip install qdrant-client
docker run -p 6333:6333 qdrant/qdrant

docker run -it \
    --rm \
    --name qdrant_image \
    -p 6333:6333 \
    -p 6334:6334 \
    -v "$(pwd)/qdrant_storage:/qdrant/storage:z" \
    qdrant/qdrant

pip install weaviate-client
docker run -p 8080:8080 semitechnologies/weaviate \
  --host 0.0.0.0 \
  --port 8080 \
  --scheme http
